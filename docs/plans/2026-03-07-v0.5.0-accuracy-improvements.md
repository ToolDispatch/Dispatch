```python
def _search_one_term(term: str, limit: int = 5) -> list:
    """Search registry for one term. Returns list of {"id", "description"}. Cached 1hr."""
    cache = _load_cache()
    registry = cache.get("registry", {})
    entry = registry.get(term, {})
    if entry and (time.time() - entry.get("fetched_at", 0)) < CACHE_TTL:
        return entry["data"]
    try:
        result = subprocess.run(
            ["npx", "--yes", "skills", "find", term],
            capture_output=True, text=True, timeout=6, check=False
        )
        lines = result.stdout.split("\n")
        skills = []
        for line in lines:
            stripped = strip_ansi(line).strip()
            if "@" not in stripped or "/" not in stripped:
                continue
            if stripped.startswith("http") or stripped.startswith("└"):
                continue
            parts = stripped.split()
            if not parts:
                continue
            skill_id = parts[0]
            if "/" in skill_id and "@" in skill_id:
                description = " ".join(parts[1:]) if len(parts) > 1 else ""
                skills.append({"id": skill_id, "description": description})
        skills = skills[:limit]
        if "registry" not in cache:
            cache["registry"] = {}
        cache["registry"][term] = {"data": skills, "fetched_at": time.time()}
        _save_cache(cache)
        return skills
    except Exception:
        return entry.get("data", [])


def search_registry(task_type: str, limit: int = 5) -> list:
    """Search skills.sh for all terms in compound task type. Returns list of {"id", "description"}.

    For "docker-aws-github-actions", searches "docker", "aws", "github-actions" separately.
    Results are deduplicated; first match wins.
    """
    terms = task_type.split("-")
    # Rebuild meaningful terms: "github-actions" is one term, not two
    # Simple heuristic: split on hyphen but re-join pairs that look like compound tool names
    # For now: take up to 3 unique terms
    seen_terms = []
    for t in terms:
        if t not in seen_terms:
            seen_terms.append(t)
    seen_terms = seen_terms[:3]

    seen_ids = set()
    results = []
    for term in seen_terms:
        for skill in _search_one_term(term, limit):
            if skill["id"] not in seen_ids:
                seen_ids.add(skill["id"])
                results.append(skill)
        if len(results) >= limit:
            break
    return results[:limit]
```

def scan_mcp_servers(cwd: str = None) -> list:
    """Scan .mcp.json files for installed MCP servers. Returns [{name, description, source, marketplace}]."""
    paths = [os.path.expanduser("~/.claude/.mcp.json")]
    if cwd:
        paths.append(os.path.join(cwd, ".mcp.json"))

    servers = []
    seen = set()
    for path in paths:
        try:
            with open(path) as f:
                data = json.load(f)
            for name, config in data.get("mcpServers", {}).items():
                if name not in seen:
                    seen.add(name)
                    description = config.get("description", f"MCP server providing {name} integration")
                    servers.append({
                        "name": name,
                        "description": description,
                        "source": "mcp",
                        "marketplace": "mcp"
                    })

```python
#!/usr/bin/env python3
"""
dispatch-test — verify Dispatch setup is working correctly

Usage: python3 dispatch-test
"""
import json
import os
import sys
import subprocess

SKILL_ROUTER_DIR = os.path.expanduser("~/.claude/skill-router")
CONFIG_FILE = os.path.join(SKILL_ROUTER_DIR, "config.json")
STATE_FILE = os.path.join(SKILL_ROUTER_DIR, "state.json")

def check(label, ok, detail=""):
    status = "OK" if ok else "FAIL"
    line = f"  [{status}] {label}"
    if detail:
        line += f": {detail}"
    print(line)
    return ok


def main():
    print("\ndispatch-test — Dispatch Setup Verification")
    print("=" * 44)

    all_ok = True

    # 1. Auth mode
    token = None
    endpoint = "https://dispatch.visionairy.biz"
    api_key = os.environ.get("ANTHROPIC_API_KEY", "")
    try:
        config = json.load(open(CONFIG_FILE))
        token = config.get("token", "")
        endpoint = config.get("endpoint", endpoint)
    except Exception:
        pass

    if token:
        mode = f"Hosted ({endpoint})"
        auth_ok = True
    elif api_key:
        mode = "BYOK (ANTHROPIC_API_KEY)"
        auth_ok = True
    else:
        mode = "NONE — Dispatch will not fire"
        auth_ok = False

    all_ok &= check("Auth mode", auth_ok, mode)

    # 2. Hook registered
    hook_path = os.path.expanduser("~/.claude/hooks/skill-router.sh")
    hook_ok = os.path.exists(hook_path)
    all_ok &= check("Hook file", hook_ok, hook_path if hook_ok else "missing — run install.sh")

    # 3. Evaluator installed
    eval_path = os.path.join(SKILL_ROUTER_DIR, "evaluator.py")
    eval_ok = os.path.exists(eval_path)
    all_ok &= check("Evaluator", eval_ok, eval_path if eval_ok else "missing — run install.sh")

    # 4. Installed plugins
    try:
        sys.path.insert(0, SKILL_ROUTER_DIR)
        from evaluator import scan_installed_plugins, get_installed_skills, scan_mcp_servers, PLUGINS_DIR
        plugins = scan_installed_plugins(PLUGINS_DIR)
        skills = get_installed_skills()
        mcps = scan_mcp_servers()
        plugin_ok = len(plugins) > 0
        all_ok &= check("Installed plugins", True, f"{len(plugins)} plugins, {len(skills)} skills, {len(mcps)} MCP servers")
        if not plugin_ok:
            print("    Tip: add marketplaces with /plugins add anthropics/claude-plugins-official")
    except Exception as e:
        all_ok &= check("Installed plugins", False, str(e))

    # 5. State file
    try:
        state = json.load(open(STATE_FILE))
        last = state.get("last_task_type", "none")
        check("State file", True, f"last task: {last}")
    except Exception:
        check("State file", True, "not found (normal on fresh install)")

    # 6. Live classify test (hosted only)
    if token:
        try:
            import urllib.request
            payload = json.dumps({
                "transcript": [],
                "cwd": os.getcwd(),
                "last_task_type": None,
                "prompt": "let me start writing some tests for this feature"
            }).encode()
            req = urllib.request.Request(
                f"{endpoint}/classify",
                data=payload,
                headers={
                    "Authorization": f"Bearer {token}",
                    "Content-Type": "application/json"
                }
            )
            with urllib.request.urlopen(req, timeout=6) as resp:
                result = json.load(resp)
                task_type = result.get("task_type", "?")
                all_ok &= check("Live classify", True, f"task_type={task_type}")
        except Exception as e:
            all_ok &= check("Live classify", False, str(e))

    # 7. BYOK classify test
    elif api_key:
        try:
            sys.path.insert(0, SKILL_ROUTER_DIR)
            from classifier import classify_topic_shift
            result = classify_topic_shift(
                ["let me start writing some tests for this feature"],
                os.getcwd(),
                api_key=api_key
            )
            task_type = result.get("task_type", "?")
            all_ok &= check("BYOK classify", True, f"task_type={task_type}")
        except Exception as e:
            all_ok &= check("BYOK classify", False, str(e))

    print()
    if all_ok:
        print("Setup looks good. Start a new CC session to activate Dispatch.")
    else:
        print("Some checks failed. Fix the issues above, then restart your CC session.")
    print()


if __name__ == "__main__":
    main()
```
